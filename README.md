```bash
├── data
│   ├── validation
│   ├── submissions
│   ├── raw
│   ├── processed
│   ├── mappings
│   └── interim
├── logs_metadata
├── models
├── notebooks
├── .gitignore
├── config.yml
├── docker-compose.yml
├── Dockerfile
├── environment.yml
├── LICENSE
├── README.md

```
